 12    """
 13    A topic-based publish-subscribe system for asynchronous message passing.
 14    """

 16    def __init__(self):
 17        self._topics: Registry = defaultdict(set)
 18        "Dictionary mapping topics to sets of subscriber queues"
 19        self._lock = asyncio.Lock()
 20        "Lock for thread-safe operations on the registry"
 21
 22    async def subscribe(self, subscriber: asyncio.Queue, *topics: str):
 23        """
 24        Subscribe a queue to one or more topics.
 25
 26        Args:
 27            subscriber (asyncio.Queue): The queue that will receive messages.
 28                This queue will receive messages as (topic, message) tuples.
 29            *topics (str): Variable number of topic strings to subscribe to.
 30                The subscriber will receive messages published to any of these topics.
 31
 32        Example:
 33            ```python
 34            queue = asyncio.Queue()
 35
 36            # Subscribe to single topic
 37            await pubsub.subscribe(queue, "temperature")
 38
 39            # Subscribe to multiple topics
 40            await pubsub.subscribe(queue, "news", "weather", "sports")
 41            ```
 42        """
 43        async with self._lock:
 44            for topic in topics:
 45                self._topics[topic].add(subscriber)
 46
 47    async def unsubscribe(self, subscriber: asyncio.Queue, *topics: str):
 48        """
 49        Unsubscribe a queue from one or more topics.
 50
 51        Args:
 52            subscriber (asyncio.Queue): The queue to unsubscribe.
 53            *topics (str): Topics to unsubscribe from.
 54
 55        Example:
 56            ```python
 57            # Unsubscribe from a single topic
 58            await pubsub.unsubscribe(queue, "weather")
 59
 60            # Unsubscribe from multiple topics
 61            await pubsub.unsubscribe(queue, "news", "sports")
 62            ```
 63        """
 64        async with self._lock:
 65            for topic in topics:
 66                subscribers = self._topics.get(topic)
 67                if subscribers is not None:
 68                    subscribers.discard(subscriber)
 69                    if not subscribers:
 70                        del self._topics[topic]
 71
 72    async def broadcast(self, message: Any, *topics: str):
 73        """
 74        Broadcast a message to all subscribers of the specified topics.
 75
 76        Args:
 77            message (Any): The message to broadcast.
 78            *topics (str): Topics to broadcast to.
 79
 80        Example:
 81            ```python
 82            await pubsub.broadcast("Hello world!", "greetings")
 83            # Subscribers receive: ("greetings", "Hello world!")
 84
 85            data = {"sensor": "temperature", "value": 23.5, "unit": "celsius"}
 86            await pubsub.broadcast(data, "telemetry", "monitoring")
 87            # Subscribers receive: ("telemetry", data)
 88            #                      ("monitoring", data)
 89            ```
 90
 91        Note:
 92            - Slow consumers may miss messages if their queue is full
 93            - The broadcast is asynchronous - it doesn't wait for subscribers to process messages
 94        """
 95        topic_subscribers = []
 96        async with self._lock:
 97            for topic in topics:
 98                subscribers = list(self._topics.get(topic, set()))
 99                if subscribers:
100                    topic_subscribers.append((topic, subscribers))
101
102        for topic, subscribers in topic_subscribers:
103            for subscriber in subscribers:
104                self._try_put_message(subscriber, topic, message)
105
106    async def broadcast_from(
107        self, publisher: asyncio.Queue, message: Any, *topics: str
108    ):
109        """
110        Broadcast a message to all subscribers except the publisher itself.
111
112        Args:
113            publisher (asyncio.Queue): The queue of the publisher to exclude.
114                This subscriber will not receive the message.
115            message (Any): The message to broadcast.
116            *topics (str): Topics to broadcast to.
117
118        Example:
119            ```python
120            publisher_queue = asyncio.Queue()
121            chat_queue = asyncio.Queue()
122            await pubsub.subscribe(publisher_queue, "chat")
123            await pubsub.subscribe(chat, "chat")
124
125            await pubsub.broadcast_from(
126                publisher_queue,
127                "User joined the channel",
128                "chat"
129            )
130            # chat_queue receives ("chat", "User joined the channel")
131            # publisher_queue doesn't receive this message
132            ```
133        """
134        topic_subscribers = []
135        async with self._lock:
136            for topic in topics:
137                subscribers = [
138                    s for s in self._topics.get(topic, set()) if s is not publisher
139                ]
140                if subscribers:
141                    topic_subscribers.append((topic, subscribers))
142
143        for topic, subscribers in topic_subscribers:
144            for subscriber in subscribers:
145                self._try_put_message(subscriber, topic, message)
146
147    def _try_put_message(self, subscriber: asyncio.Queue, topic: str, message: Any):
148        try:
149            subscriber.put_nowait((topic, message))
150        except (asyncio.QueueFull, asyncio.QueueShutDown):
151            # Handle slow consumers – you might want to drop or log
152            pass
