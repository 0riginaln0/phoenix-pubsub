 36class PubSub:
 37    """
 38    A topic-based publish-subscribe system for asynchronous message passing.
 39
 40    Supports attaching arbitrary metadata to subscriptions, allowing custom
 41    dispatcher logic for flexible message routing.
 42    """

 44    def __init__(self):
 45        self._topics: Registry = defaultdict(dict)
 46        "Dictionary mapping Topics to Subscribers"
 47        self._lock = asyncio.Lock()
 48        "Lock for thread-safe operations on the registry"
 49
 50    async def subscribe(
 51        self, peer: Peer, *topics: str, metadata: Optional[dict] = None
 52    ):
 53        """
 54        Subscribe a queue to one or more topics.
 55
 56        Args:
 57            peer: The queue that will receive messages.
 58                This queue will receive messages as (topic, message) tuples.
 59            *topics: Variable number of topic strings to subscribe to.
 60                The subscriber will receive messages published to any of these topics.
 61            metadata: Arbitrary data attached to this subscription.
 62                The dispatcher can use it for custom delivery logic (e.g., filtering,
 63                prioritisation).
 64
 65        Example:
 66            ```python
 67            queue = asyncio.Queue()
 68
 69            # Subscribe to single topic
 70            await pubsub.subscribe(queue, "temperature")
 71
 72            # Subscribe to multiple topics
 73            await pubsub.subscribe(queue, "news", "weather", "sports")
 74
 75            # Provide metadata
 76            await pubsub.subscribe(queue, "stock", metadata={"priority": 10})
 77            ```
 78        """
 79        metadata = metadata or {}
 80        async with self._lock:
 81            for topic in topics:
 82                self._topics[topic][peer] = metadata
 83
 84    async def unsubscribe(self, peer: Peer, *topics: str):
 85        """
 86        Unsubscribe a queue from one or more topics.
 87
 88        Args:
 89            peer: The queue to unsubscribe.
 90            *topics: Topics to unsubscribe from.
 91
 92        Example:
 93            ```python
 94            # Unsubscribe from a single topic
 95            await pubsub.unsubscribe(queue, "weather")
 96
 97            # Unsubscribe from multiple topics
 98            await pubsub.unsubscribe(queue, "news", "sports")
 99            ```
100        """
101        async with self._lock:
102            for topic in topics:
103                if subscribers := self._topics.get(topic):
104                    subscribers.pop(peer, None)
105                    if not subscribers:
106                        del self._topics[topic]

108    async def broadcast(
109        self,
110        message: Message,
111        *topics: str,
112        dispatcher: Dispatcher = default_dispatcher,
113    ):
114        """
115        Broadcast a message to all subscribers of the specified topics.
116
117        Args:
118            message: The message to broadcast.
119            *topics: Topics to broadcast to.
120            dispatcher: A callable that handles the actual message delivery.
121                Custom dispatchers can implement filtering, transformation,
122                or prioritisation using the metadata stored with each subscriber.
123
124        Example:
125            ```python
126            await pubsub.broadcast("Hello world!", "greetings")
127            # Subscribers receive: ("greetings", "Hello world!")
128
129            data = {"sensor": "temperature", "value": 23.5, "unit": "celsius"}
130            await pubsub.broadcast(data, "telemetry", "monitoring")
131            # Subscribers receive: ("telemetry", data)
132            #                      ("monitoring", data)
133            ```
134
135        Note:
136            - Slow consumers may miss messages if their queue is full
137            - The broadcast is asynchronous - it doesn't wait for subscribers to process messages
138        """
139        pending: list[tuple[Topic, dict[Peer, dict]]] = []
140        async with self._lock:
141            for topic in topics:
142                if subscribers := self._topics.get(topic):
143                    pending.append((topic, subscribers.copy()))
144
145        for topic, subscribers in pending:
146            dispatcher(topic, message, subscribers, None)

149        self,
150        publisher: Peer,
151        message: Message,
152        *topics: str,
153        dispatcher: Dispatcher = default_dispatcher,
154    ):
155        """
156        Broadcast a message to all subscribers except the publisher itself.
157
158        Args:
159            publisher: The queue of the publisher to exclude.
160                This peer (the publisher) will not receive the message.
161            message: The message to broadcast.
162            *topics: Topics to broadcast to.
163            dispatcher: A callable that handles the actual message delivery.
164                Custom dispatchers can implement filtering, transformation,
165                or prioritisation using the metadata stored with each subscriber.
166
167        Example:
168            ```python
169            publisher_queue = asyncio.Queue()
170            chat_queue = asyncio.Queue()
171            await pubsub.subscribe(publisher_queue, "chat")
172            await pubsub.subscribe(chat_queue, "chat")
173
174            await pubsub.broadcast_from(
175                publisher_queue,
176                "User joined the channel",
177                "chat"
178            )
179            # chat_queue receives ("chat", "User joined the channel")
180            # publisher_queue doesn't receive this message
181            ```
182        """
183        pending: list[tuple[Topic, dict[Peer, dict]]] = []
184        async with self._lock:
185            for topic in topics:
186                subscribers = self._topics.get(topic)
187                if subscribers:
188                    pending.append((topic, subscribers.copy()))
189
190        for topic, subscribers in pending:
191            dispatcher(topic, message, subscribers, publisher)
